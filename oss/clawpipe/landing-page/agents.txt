product:
  name: ClawPipe
  tagline: The only AI gateway that skips LLM calls entirely.
  homepage: https://clawpipe.ai
  status: GA
  pricing: https://clawpipe.ai/#pricing

actions:
  - id: send-prompt
    summary: Route a prompt through the ClawPipe pipeline (Booster -> Pack -> Cache -> Route -> Call -> Learn).
    method: POST
    url: https://api.clawpipe.ai/v1/prompt
    auth: Bearer cp_live_... (get one at https://app.clawpipe.ai/signup — free tier 1k calls/day)
    body_schema: https://clawpipe.ai/openapi.json#/components/schemas/PromptRequest
    example_body: |
      { "prompt": "Translate to French: Hello", "provider": "openai", "model": "gpt-4o-mini" }
    headers:
      Authorization: required
      X-Project-Id: required
      Idempotency-Key: optional, [A-Za-z0-9_.-]{1,200}
    response_headers:
      RateLimit-Limit, RateLimit-Remaining, RateLimit-Reset (RFC 9239)

  - id: get-savings
    summary: Read per-project saved-USD aggregates (this month + lifetime + percent).
    method: GET
    url: https://api.clawpipe.ai/v1/savings
    auth: Bearer

  - id: get-public-stats
    summary: Anonymized aggregate stats across all projects. No auth.
    method: GET
    url: https://api.clawpipe.ai/v1/index

  - id: discover
    summary: Self-served OpenAPI 3.1 spec. No auth.
    method: GET
    url: https://api.clawpipe.ai/v1/openapi.json
