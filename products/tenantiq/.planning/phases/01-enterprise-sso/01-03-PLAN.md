<!-- INSERT OR IGNORE + re-fetch pattern — from RESEARCH.md Pattern 4 -->
```typescript
// Step 1: try to find existing user
const existing = await db.prepare(
  'SELECT id FROM platform_users WHERE email = ? AND organization_id = ?'
).bind(email, orgId).first<{ id: string }>();
if (existing) return existing.id;

// Step 2: insert (OR IGNORE in case of concurrent race)
const id = crypto.randomUUID();
await db.prepare(`
  INSERT OR IGNORE INTO platform_users
    (id, organization_id, email, display_name, role, auth_provider, scope_level, created_at)
  VALUES (?, ?, ?, ?, ?, 'sso', 'org', ?)
`).bind(id, orgId, email, name ?? email, role ?? 'member', Date.now()).run();

// Step 3: re-fetch to get canonical ID (handles race where OR IGNORE fired)
const row = await db.prepare(
  'SELECT id FROM platform_users WHERE email = ? AND organization_id = ?'
).bind(email, orgId).first<{ id: string }>();
return row!.id;
```
