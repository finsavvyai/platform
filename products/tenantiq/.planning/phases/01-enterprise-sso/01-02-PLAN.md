<!-- KV nonce pattern â€” mirror from auth-callback.ts -->
```typescript
// Store nonce on login initiation
await c.env.KV.put(
  `sso:state:${nonce}`,
  JSON.stringify({ orgId: conn.org_id, domain, connId: conn.id }),
  { expirationTtl: 300 }  // SSO-06: exactly 300 seconds
);

// Consume nonce on callback (one-time use)
const stored = await c.env.KV.get(`sso:state:${state}`);
if (!stored) return c.json({ error: 'invalid or expired state' }, 400);
await c.env.KV.delete(`sso:state:${state}`);
const { orgId, connId } = JSON.parse(stored);
```
