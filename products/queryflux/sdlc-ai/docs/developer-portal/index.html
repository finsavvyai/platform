Quick Start Example

// Install the SDK
npm install @sdlc.ai/sdk

// Initialize the client
import { SDLCClient } from '@sdlc.ai/sdk';

const client = new SDLCClient({
  apiKey: 'your-api-key-here',
  baseUrl: 'https://api.sdlc.ai/v3'
});

// Upload a document
const document = await client.documents.upload({
  file: './my-document.pdf',
  name: 'Important Document',
  description: 'Company policy document'
});

// Query with RAG
const response = await client.rag.query({
  query: 'What is the company policy on remote work?',
  documentIds: [document.id]
});

console.log(response.answer);
console.log(response.sources);

package main

import (
    "context"
    "fmt"
    "log"

    "github.com/sdlc-ai/sdlc-go"
)

func main() {
    client := sdlc.NewClient("your-api-key-here")

    // Upload a document
    doc, err := client.Documents.Upload(context.Background(), &sdlc.UploadDocumentRequest{
        FilePath: "./my-document.pdf",
        Name:     "Important Document",
        Description: "Company policy document",
    })
    if err != nil {
        log.Fatal(err)
    }

    // Query with RAG
    response, err := client.RAG.Query(context.Background(), &sdlc.QueryRequest{
        Query:       "What is the company policy on remote work?",
        DocumentIDs: []string{doc.ID},
    })
    if err != nil {
        log.Fatal(err)
    }

    fmt.Println(response.Answer)
    for _, source := range response.Sources {
        fmt.Printf("- %s (Score: %.2f)\n", source.DocumentName, source.Score)
    }
}

Document Processing Pipeline

Upload and process multiple documents with progress tracking

Show Code 
async function processDocuments(files) {
  const results = [];

  for (const file of files) {
    // Upload document
    const document = await client.documents.upload({
      file: file,
      description: `Uploaded at ${new Date().toISOString()}`
    });

    // Poll for processing completion
    let status = await client.documents.getStatus(document.id);
    while (status.status === 'processing') {
      await new Promise(resolve => setTimeout(resolve, 1000));
      status = await client.documents.getStatus(document.id);
    }

    results.push({
      id: document.id,
      name: file.name,
      status: status.status,
      chunks: status.metadata.chunks
    });
  }

  return results;
}

Advanced RAG Query

async def advanced_rag_query(query, filters=None):
    # Retrieve context with custom settings
    context = await client.rag.get_context(
        query=query,
        context_limit=6000,  # tokens
        document_filters=filters,
        min_score=0.7,
        max_sources=5
    )

    # Generate response with custom prompt
    response = await client.llm.chat(
        messages=[
            {
                "role": "system",
                "content": "You are a helpful assistant. Use the provided context to answer accurately."
            },
            {
                "role": "user",
                "content": f"Context:\n{context.context}\n\nQuestion: {query}"
            }
        ],
        model="gpt-4",
        temperature=0.3
    )

    return {
        "answer": response.content,
        "sources": context.sources,
        "tokens_used": response.usage.total_tokens
    }

Hybrid Vector Search

Combine semantic and keyword search for better results

Show Code 
func HybridSearch(ctx context.Context, query string, limit int) ([]SearchResult, error) {
    // Get query embedding
    embedding, err := client.LLM.GetEmbedding(ctx, query)
    if err != nil {
        return nil, err
    }

    // Perform vector search
    vectorResults, err := client.Vector.Search(ctx, &SearchRequest{
        Vector:   embedding,
        TopK:     limit,
        Filters:  map[string]interface{}{"status": "processed"},
    })
    if err != nil {
        return nil, err
    }

    // Perform keyword search
    keywordResults, err := client.Documents.Search(ctx, &SearchDocumentsRequest{
        Query: query,
        Limit: limit,
    })
    if err != nil {
        return nil, err
    }

    // Merge and rank results
    return mergeAndRank(vectorResults, keywordResults), nil
}

DLP-Aware Document Upload

Scan documents for sensitive data before upload

Show Code 
async function secureDocumentUpload(file) {
    // Extract text first
    const extractedText = await client.documents.extractText(file);

    // Scan for sensitive data
    const scanResult = await client.dlp.scan({
        text: extractedText,
        infoTypes: ['EMAIL', 'PHONE', 'CREDIT_CARD', 'SSN']
    });

    // Check if sensitive data found
    if (scanResult.findings.length > 0) {
        // Option 1: Redact and upload
        const redactedText = await client.dlp.redact({
            text: extractedText,
            findings: scanResult.findings
        });

        return await client.documents.upload({
            file: redactedText,
            name: file.name,
            redacted: true
        });
    } else {
        // Safe to upload directly
        return await client.documents.upload({
            file: file,
            name: file.name
        });
    }
}
